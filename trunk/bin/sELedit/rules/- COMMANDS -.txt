################################################
##### COMMANDS DESCRIPTION FOR RULES FILES #####
################################################
#
# REMOVELIST:0
# remove a list | 0 = list index
#
# REPLACEOFFSET:0|xx
# Replace the offset value of a list with the given hex bytes | 0 = list index, xx = hex value example: A3-87-C7-BE
#
# REMOVEVALUE:0:1
# remove a value from a list | 0 = list index, 1 = field index
